```python
# backend/scripts/rag_eval/types.py
"""RAG 评测台架的核心数据类型(纯数据 + 序列化,无重依赖)。"""
from __future__ import annotations

import hashlib
import json
from dataclasses import asdict, dataclass, field
from pathlib import Path


@dataclass(frozen=True)
class GoldSpan:
    ingest_record_id: int
    doc_char_start: int
    doc_char_end: int


@dataclass
class GoldenItem:
    id: str
    question: str
    gold_spans: list[GoldSpan]
    reference_answer: str
    slices: dict
    provenance: str            # synthetic | hand | benchmark
    source: str                # own | benchmark:<name> | synthetic-doc
    corpus_version: str

    def to_dict(self) -> dict:
        d = asdict(self)
        return d

    @classmethod
    def from_dict(cls, d: dict) -> "GoldenItem":
        spans = [GoldSpan(**s) for s in d["gold_spans"]]
        return cls(
            id=d["id"], question=d["question"], gold_spans=spans,
            reference_answer=d["reference_answer"], slices=d["slices"],
            provenance=d["provenance"], source=d["source"],
            corpus_version=d["corpus_version"],
        )


@dataclass
class RunConfig:
    retrieval: dict = field(default_factory=dict)
    chunker: dict = field(default_factory=dict)
    generation: dict = field(default_factory=dict)
    eval: dict = field(default_factory=dict)

    def _hash(self, *parts: dict) -> str:
        blob = json.dumps(parts, sort_keys=True, ensure_ascii=False)
        return hashlib.sha256(blob.encode("utf-8")).hexdigest()[:12]

    def config_hash(self) -> str:
        return self._hash(self.retrieval, self.chunker, self.generation, self.eval)

    def chunker_hash(self) -> str:
        return self._hash(self.chunker)


def dump_golden(items: list[GoldenItem], path: Path) -> None:
    path = Path(path)
    path.parent.mkdir(parents=True, exist_ok=True)
    with path.open("w", encoding="utf-8") as f:
        for it in items:
            f.write(json.dumps(it.to_dict(), ensure_ascii=False) + "\n")


def load_golden(path: Path) -> list[GoldenItem]:
    out = []
    for line in Path(path).read_text(encoding="utf-8").splitlines():
        line = line.strip()
        if line:
            out.append(GoldenItem.from_dict(json.loads(line)))
    return out
```

```python
# backend/scripts/rag_eval/metrics_retrieval.py
"""确定性检索指标(免 LLM,内循环)。命中=同 ingest_record_id 且 char 区间半开重叠。"""
from __future__ import annotations

from math import log2

from scripts.rag_eval.types import GoldSpan


def _overlaps(a0: int, a1: int, b0: int, b1: int) -> bool:
    return a0 < b1 and b0 < a1            # 半开区间 [start,end);相接不算


def chunk_hits_any(chunk, gold_spans: list[GoldSpan]) -> bool:
    for g in gold_spans:
        if chunk.ingest_record_id == g.ingest_record_id and \
                _overlaps(chunk.char_start, chunk.char_end, g.doc_char_start, g.doc_char_end):
            return True
    return False


def recall_at_k(ranked, gold_spans: list[GoldSpan], k: int) -> float:
    """any-hit:top-k 里有任一命中任一 gold → 1.0,否则 0.0。"""
    return 1.0 if any(chunk_hits_any(c, gold_spans) for c in ranked[:k]) else 0.0


def coverage_at_k(ranked, gold_spans: list[GoldSpan], k: int) -> float:
    """多跳覆盖率:被 top-k 命中的 gold 跨度数 / 总 gold 跨度数。"""
    if not gold_spans:
        return 0.0
    top = ranked[:k]
    covered = sum(1 for g in gold_spans if any(chunk_hits_any(c, [g]) for c in top))
    return covered / len(gold_spans)


def mrr(ranked, gold_spans: list[GoldSpan]) -> float:
    for i, c in enumerate(ranked, start=1):
        if chunk_hits_any(c, gold_spans):
            return 1.0 / i
    return 0.0


def ndcg_at_k(ranked, gold_spans: list[GoldSpan], k: int) -> float:
    gains = [1.0 if chunk_hits_any(c, gold_spans) else 0.0 for c in ranked[:k]]
    dcg = sum(g / log2(i + 1) for i, g in enumerate(gains, start=1))
    ideal = sorted(gains, reverse=True)
    idcg = sum(g / log2(i + 1) for i, g in enumerate(ideal, start=1))
    return dcg / idcg if idcg > 0 else 0.0


def context_precision_at_k(ranked, gold_spans: list[GoldSpan], k: int) -> float:
    top = ranked[:k]
    if not top:
        return 0.0
    return sum(1 for c in top if chunk_hits_any(c, gold_spans)) / len(top)


def context_precision_ordered(ranked, gold_spans: list[GoldSpan], k: int) -> float:
    """RAGAS 式有序版:Σ(precision@i × hit_i) / 命中数;命中越靠前得分越高。"""
    top = ranked[:k]
    hits = 0
    weighted = 0.0
    for i, c in enumerate(top, start=1):
        if chunk_hits_any(c, gold_spans):
            hits += 1
            weighted += hits / i
    return weighted / hits if hits else 0.0
```

```python
# backend/scripts/rag_eval/metrics_citation.py
"""确定性引用指标。忠实度(需 judge)在 metrics_generation。"""
from __future__ import annotations

import re

from scripts.rag_eval.types import GoldSpan
from scripts.rag_eval.metrics_retrieval import chunk_hits_any

_CITE = re.compile(r"\[(\d+)\]")


class _Cite:
    """把 citation dict 适配成 chunk_hits_any 需要的 .ingest_record_id/.char_start/.char_end。"""
    def __init__(self, c: dict) -> None:
        self.ingest_record_id = c["ingest_record_id"]
        self.char_start = c["char_start"]
        self.char_end = c["char_end"]


def parse_citation_ids(answer: str) -> list[int]:
    seen, out = set(), []
    for m in _CITE.findall(answer):
        n = int(m)
        if n not in seen:
            seen.add(n)
            out.append(n)
    return out


def citation_validity(answer: str, n_chunks: int) -> float:
    ids = parse_citation_ids(answer)
    if not ids:
        return 1.0
    valid = sum(1 for n in ids if 1 <= n <= n_chunks)
    return valid / len(ids)


def citation_accuracy(citations: list[dict], gold_spans: list[GoldSpan]) -> float:
    if not citations:
        return 0.0
    hit = sum(1 for c in citations if chunk_hits_any(_Cite(c), gold_spans))
    return hit / len(citations)
```

```python
# backend/scripts/rag_eval/judge.py
"""LLM judge 封装:结构化 JSON + 磁盘缓存 + 失败回 None(=NaN)。判不出≠不忠实。"""
from __future__ import annotations

import hashlib
import json
import re
from pathlib import Path

_FENCE = re.compile(r"```(?:json)?\s*(\{.*?\})\s*```", re.DOTALL)
_OBJ = re.compile(r"\{.*\}", re.DOTALL)


def _extract_json(text: str) -> dict | None:
    for pat in (_FENCE, _OBJ):
        m = pat.search(text)
        if m:
            try:
                return json.loads(m.group(1) if pat is _FENCE else m.group(0))
            except json.JSONDecodeError:
                continue
    return None


def _h(s: str) -> str:
    return hashlib.sha256(s.encode("utf-8")).hexdigest()[:16]


class Judge:
    def __init__(self, llm, model_name: str, cache_path: Path) -> None:
        self._llm = llm
        self._model = model_name
        self._cache_path = Path(cache_path)
        self._cache = self._load_cache()

    def _load_cache(self) -> dict:
        if self._cache_path.is_file():
            try:
                return json.loads(self._cache_path.read_text(encoding="utf-8"))
            except (json.JSONDecodeError, OSError):
                return {}
        return {}

    def _save_cache(self) -> None:
        self._cache_path.parent.mkdir(parents=True, exist_ok=True)
        tmp = self._cache_path.with_suffix(".tmp")
        tmp.write_text(json.dumps(self._cache, ensure_ascii=False), encoding="utf-8")
        tmp.replace(self._cache_path)

    def ask(self, metric: str, item_id: str, messages: list[dict], *,
            answer: str = "", context: str = "") -> dict | None:
        key = "|".join([metric, item_id, _h(answer), _h(context), self._model])
        if key in self._cache:
            return self._cache[key]
        try:
            raw = self._llm.complete(messages)
        except Exception:  # noqa: BLE001 — judge 调用故障 → None(NaN),绝不当 0
            return None
        parsed = _extract_json(raw)
        if parsed is None:
            return None
        self._cache[key] = parsed
        self._save_cache()
        return parsed
```

```python
# backend/scripts/rag_eval/corpus.py
"""从只读源目录拷一份分层切片到冻结 eval-data + manifest。绝不写源目录。"""
from __future__ import annotations

import hashlib
import json
import shutil
from pathlib import Path

TEXT_EXTS = {"md", "txt", "pdf", "docx", "pptx", "html", "htm", "srt",
             "py", "java", "c", "h", "cpp", "json", "csv"}


def sha256_file(path: Path) -> str:
    h = hashlib.sha256()
    with Path(path).open("rb") as f:
        for chunk in iter(lambda: f.read(65536), b""):
            h.update(chunk)
    return h.hexdigest()


def _eligible(p: Path) -> bool:
    if not p.is_file():
        return False
    ext = p.suffix.lower().lstrip(".")
    return ext in TEXT_EXTS and not p.name.startswith(".")


def build_corpus(sources: list[dict], dest: Path) -> list[dict]:
    dest = Path(dest)
    rows: list[dict] = []
    for spec in sources:
        src_root = Path(spec["path"])
        slices = spec["slices"]
        domain = slices.get("domain", "misc")
        for p in sorted(src_root.rglob("*")):
            if not _eligible(p):
                continue
            rel = f"{domain}/{p.name}"
            out = dest / rel
            out.parent.mkdir(parents=True, exist_ok=True)
            # 重名加序号,避免覆盖
            i = 1
            while out.exists():
                out = dest / f"{domain}/{p.stem}__{i}{p.suffix}"
                rel = str(out.relative_to(dest))
                i += 1
            shutil.copy2(p, out)                 # 拷贝(只读源 → 冻结副本),绝不改源
            rows.append({"rel_path": rel, "sha256": sha256_file(out),
                         "source_path": str(p), "slices": slices})
    man = dest / "manifest.jsonl"
    man.parent.mkdir(parents=True, exist_ok=True)
    with man.open("w", encoding="utf-8") as f:
        for r in rows:
            f.write(json.dumps(r, ensure_ascii=False) + "\n")
    return rows
```

```python
"""按 chunker-hash 归档的冻结索引快照。真索引构建经 ingest_fn 注入(测试用 fake)。"""
from __future__ import annotations

from pathlib import Path


def snapshot_dir(base: Path, chunker_hash: str) -> Path:
    return Path(base) / chunker_hash


def _marker(base: Path, chunker_hash: str) -> Path:
    return snapshot_dir(base, chunker_hash) / ".ready"


def snapshot_ready(base: Path, chunker_hash: str) -> bool:
    return _marker(base, chunker_hash).is_file()


def build_snapshot(base: Path, chunker_hash: str, ingest_fn) -> Path:
    d = snapshot_dir(base, chunker_hash)
    if snapshot_ready(base, chunker_hash):
        return d
    d.mkdir(parents=True, exist_ok=True)
    ingest_fn(d)                                   # 失败则抛出,不写 .ready
    _marker(base, chunker_hash).write_text("ok", encoding="utf-8")
    return d
```

```python
"""两层 runner:point ① 检索器单测,point ②③ 端到端(真 ChatService 经注入)。"""
from __future__ import annotations

import json
from dataclasses import dataclass
from pathlib import Path

from scripts.rag_eval.types import GoldenItem, RunConfig
from scripts.rag_eval import metrics_retrieval as mr
from scripts.rag_eval import metrics_citation as mc
from scripts.rag_eval import metrics_generation as mg


class RecordingRetriever:
    """包真 HybridRetriever,记录本轮每次 retrieve 的结果(point ② 的合并池)。"""
    def __init__(self, inner) -> None:
        self._inner = inner
        self.seen: list[list] = []

    def retrieve(self, **kwargs):
        out = self._inner.retrieve(**kwargs)
        self.seen.append(out)
        return out

    def pool(self) -> list:
        flat = []
        for lst in self.seen:
            flat.extend(lst)
        return flat


@dataclass
class GenOutput:
    answer: str
    pool: list
    citations: list[dict]


def _retrieval_metrics(ranked, gold_spans, k_values) -> dict:
    out: dict = {"mrr": mr.mrr(ranked, gold_spans)}
    for k in k_values:
        out[f"recall@{k}"] = mr.recall_at_k(ranked, gold_spans, k)
        out[f"coverage@{k}"] = mr.coverage_at_k(ranked, gold_spans, k)

    recs = []

        ranked = retrieve_fn(it.question)

                     "retrieval": _retrieval_metrics(ranked, it.gold_spans, k_values)})

def evaluate_full(items: list[GoldenItem], generate_fn, judge, k_values) -> list[dict]:
    recs = []
    for it in items:
        g: GenOutput = generate_fn(it.question)
        unanswerable = it.slices.get("q_type") == "negation"
        context = "\n".join(getattr(c, "text", "") for c in g.pool)
        rec = {
            "item_id": it.id, "slices": it.slices, "answer": g.answer,
            "retrieval": _retrieval_metrics(g.pool, it.gold_spans, k_values),
            "citation": {
                "citation_validity": mc.citation_validity(g.answer, len(g.pool)),
                "citation_accuracy": mc.citation_accuracy(g.citations, it.gold_spans),
                "citation_faithfulness": mg.citation_faithfulness(judge, it.id, g.answer, g.citations),
            },
            "generation": {
                "faithfulness": mg.faithfulness(judge, it.id, g.answer, context),
                "answer_relevancy": mg.answer_relevancy(judge, it.id, it.question, g.answer),
                "answer_correctness": mg.answer_correctness(judge, it.id, g.answer, it.reference_answer),
                "refusal_correctness": mg.refusal_correctness(judge, it.id, g.answer, unanswerable),
            },
        }
        recs.append(rec)
    return recs

    out_base = Path(out_base)
```

```python
# backend/scripts/rag_eval/report.py
"""分片 × 分指标报告 + run-vs-run delta。NaN(judge 失败)从均值剔除。"""
from __future__ import annotations

from math import isnan


def _get(rec: dict, path: str):
    cur = rec
    for part in path.split("."):
        cur = cur.get(part) if isinstance(cur, dict) else None
        if cur is None:
            return None
    return cur


def _mean(vals: list[float]) -> float:
    clean = [v for v in vals if v is not None and not (isinstance(v, float) and isnan(v))]
    return sum(clean) / len(clean) if clean else float("nan")


def overall(records: list[dict], metric_path: str) -> float:
    return _mean([_get(r, metric_path) for r in records])


def aggregate(records: list[dict], metric_path: str, group_by: str) -> dict:
    groups: dict[str, list] = {}
    for r in records:
        key = str(r.get("slices", {}).get(group_by, "?"))
        groups.setdefault(key, []).append(_get(r, metric_path))
    return {k: _mean(v) for k, v in groups.items()}


def diff_summaries(a: dict, b: dict) -> dict:
    keys = set(a) | set(b)
    return {k: (b.get(k, 0.0) - a.get(k, 0.0)) for k in sorted(keys)}


def format_markdown(records: list[dict], metric_paths: list[str]) -> str:
    lines = ["# RAG eval report", "", f"items: {len(records)}", ""]
    for mp in metric_paths:
        lines.append(f"## {mp}")
        lines.append(f"- overall: {overall(records, mp):.3f}")
        for dim in ("domain", "doc_type", "lang", "q_type"):
            agg = aggregate(records, mp, dim)
            if len(agg) > 1 or (agg and "?" not in agg):
                cells = "  ".join(f"{k}={v:.3f}" for k, v in sorted(agg.items()))
                lines.append(f"- by {dim}: {cells}")
        lines.append("")
    return "\n".join(lines)
```

```python
"""合成 golden:从 chunk 生成题+参考答案+支撑句;gold 收窄到支撑句偏移 + 过滤泄漏/指代。"""
from __future__ import annotations

from scripts.rag_eval.types import GoldenItem, GoldSpan

_GEN_SYS = (
    "你是出题人。仅依据【片段】出一道**只能由它回答**的自然中文问题,并给参考答案,"
    "再原文摘抄一句支撑答案的话(support_quote 必须是片段原文子串)。"
    '只输出 JSON:{"question":"...","reference_answer":"...","support_quote":"..."}。')

_DEICTIC = ("这段", "上文", "下文", "本段", "如图", "上图", "下图", "该文", "这篇", "此处", "上述")
_LEAK_NGRAM = 8        # 问题里出现原文连续 >=8 字的子串 → 判泄漏(可调)


def parse_generated(raw: str) -> dict | None:
    d = _extract_json(raw)
    if not d or not all(k in d for k in ("question", "reference_answer", "support_quote")):
        return None
    return d


def is_self_contained(question: str) -> bool:
    return not any(d in question for d in _DEICTIC)


def is_leaky(question: str, chunk_text: str) -> bool:
    for i in range(0, max(0, len(question) - _LEAK_NGRAM) + 1):
        if question[i:i + _LEAK_NGRAM] in chunk_text:
            return True
    return False


def gen_item_for_chunk(llm, chunk: dict, slices: dict, corpus_version: str) -> GoldenItem | None:
    raw = llm.complete([{"role": "system", "content": _GEN_SYS},
                        {"role": "user", "content": f"【片段】\n{chunk['doc_text']}"}])
    d = parse_generated(raw)
    if d is None:
        return None
    q, ref, quote = d["question"], d["reference_answer"], d["support_quote"]
    if not is_self_contained(q) or is_leaky(q, chunk["doc_text"]):
        return None
    pos = chunk["doc_text"].find(quote)
    if pos < 0 or not quote.strip():
        return None                       # 支撑句不在原文 → 弃(防幻觉标注)
    base = chunk["char_start"]
    span = GoldSpan(ingest_record_id=chunk["ingest_record_id"],
                    doc_char_start=base + pos, doc_char_end=base + pos + len(quote))
    qid = "g" + json.dumps([chunk["ingest_record_id"], span.doc_char_start],
                           sort_keys=True)
    return GoldenItem(id=qid, question=q, gold_spans=[span], reference_answer=ref,
                      slices=slices, provenance="synthetic", source="own",
                      corpus_version=corpus_version)
```

```python
# backend/scripts/rag_eval/calibrate.py
"""judge 可信前提:judge 裁决 vs 人工标注的 Cohen's kappa。"""
from __future__ import annotations

from math import nan


def cohens_kappa(human: list, machine: list) -> float:
    if len(human) != len(machine) or not human:
        return nan
    n = len(human)
    po = sum(1 for h, m in zip(human, machine) if h == m) / n
    labels = set(human) | set(machine)
    pe = 0.0
    for lab in labels:
        ph = sum(1 for h in human if h == lab) / n
        pm = sum(1 for m in machine if m == lab) / n
        pe += ph * pm
    if pe >= 1.0:                       # 退化(全同一类)→ 无意义
        return nan
    return (po - pe) / (1 - pe)
```

```python
        return [json.loads(l) for l in

    a, b = load(args.run_a), load(args.run_b)
```
